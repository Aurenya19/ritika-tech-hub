1. Arrays & Two Pointer Technique

üìñ What You'll Learn

Arrays are the most fundamental data structure. They store elements in contiguous memory locations.

Two Pointer Technique is a powerful pattern used to solve array problems efficiently in O(n) time.

  • When to use: Sorted arrays, finding pairs, removing duplicates
  • Time Complexity: O(n) instead of O(n¬≤)
  • Space Complexity: O(1) - no extra space needed

üí° Concept Explanation

Two Pointer Pattern:

  • Start with two pointers: one at beginning (left), one at end (right)
  • Move pointers based on condition
  • Continue until pointers meet

Example Problem: Find two numbers in sorted array that sum to target.

Approach:

  • If sum < target ‚Üí move left pointer right (increase sum)
  • If sum > target ‚Üí move right pointer left (decrease sum)
  • If sum == target ‚Üí found answer!
// Two Sum Problem - Sorted Array
function twoSum(arr, target) {
    let left = 0;
    let right = arr.length - 1;
    
    while (left < right) {
        let sum = arr[left] + arr[right];
        
        if (sum === target) {
            return [left, right];
        } else if (sum < target) {
            left++;  // Need bigger sum
        } else {
            right--; // Need smaller sum
        }
    }
    
    return [-1, -1]; // Not found
}

// Example Usage
let arr = [1, 2, 3, 4, 6, 8, 10];
let target = 10;
let result = twoSum(arr, target);
console.log(result); // Output: [1, 5] (arr[1]=2, arr[5]=8)

// Explanation:
// Start: left=0 (arr[0]=1), right=6 (arr[6]=10)
// Sum = 1+10 = 11 > 10, so right--
// Now: left=0 (arr[0]=1), right=5 (arr[5]=8)
// Sum = 1+8 = 9 < 10, so left++
// Now: left=1 (arr[1]=2), right=5 (arr[5]=8)
// Sum = 2+8 = 10 ‚úì Found!

Problem 2: Remove Duplicates from Sorted Array

function removeDuplicates(arr) {
    if (arr.length === 0) return 0;
    
    let i = 0; // Slow pointer
    
    for (let j = 1; j < arr.length; j++) { // Fast pointer
        if (arr[j] !== arr[i]) {
            i++;
            arr[i] = arr[j];
        }
    }
    
    return i + 1; // New length
}

// Example
let arr = [1, 1, 2, 2, 2, 3, 4, 4, 5];
let newLength = removeDuplicates(arr);
console.log(newLength); // 5
console.log(arr.slice(0, newLength)); // [1, 2, 3, 4, 5]
2. Sliding Window Technique

üìñ Concept

Sliding Window is used for problems involving subarrays or substrings.

Key Idea: Maintain a "window" that slides through the array.

  • Fixed Size Window: Window size remains constant
  • Variable Size Window: Window expands/shrinks based on condition
// Maximum Sum Subarray of Size K
function maxSumSubarray(arr, k) {
    let maxSum = 0;
    let windowSum = 0;
    
    // Calculate sum of first window
    for (let i = 0; i < k; i++) {
        windowSum += arr[i];
    }
    maxSum = windowSum;
    
    // Slide the window
    for (let i = k; i < arr.length; i++) {
        windowSum = windowSum - arr[i - k] + arr[i];
        maxSum = Math.max(maxSum, windowSum);
    }
    
    return maxSum;
}

// Example
let arr = [2, 1, 5, 1, 3, 2];
let k = 3;
console.log(maxSumSubarray(arr, k)); // 9 (5+1+3)

// Step by step:
// Window 1: [2,1,5] = 8
// Window 2: [1,5,1] = 7 (remove 2, add 1)
// Window 3: [5,1,3] = 9 (remove 1, add 3) ‚úì Max
// Window 4: [1,3,2] = 6 (remove 5, add 2)
3. Binary Search

üìñ Concept

Binary Search is the most efficient searching algorithm for sorted arrays.

Time Complexity: O(log n) - Much faster than linear search O(n)

How it works:

  • Start with middle element
  • If target < middle ‚Üí search left half
  • If target > middle ‚Üí search right half
  • Repeat until found or no elements left
function binarySearch(arr, target) {
    let left = 0;
    let right = arr.length - 1;
    
    while (left <= right) {
        let mid = Math.floor((left + right) / 2);
        
        if (arr[mid] === target) {
            return mid; // Found!
        } else if (arr[mid] < target) {
            left = mid + 1; // Search right half
        } else {
            right = mid - 1; // Search left half
        }
    }
    
    return -1; // Not found
}

// Example
let arr = [1, 3, 5, 7, 9, 11, 13, 15];
console.log(binarySearch(arr, 7));  // 3
console.log(binarySearch(arr, 10)); // -1

// Visualization for target=7:
// Step 1: [1,3,5,7,9,11,13,15] mid=9 (too big)
// Step 2: [1,3,5,7] mid=5 (too small)
// Step 3: [7] mid=7 ‚úì Found at index 3!

1. Introduction to Machine Learning

üìñ What is Machine Learning?

Machine Learning is teaching computers to learn from data without explicit programming.

Types of ML:

  • Supervised Learning: Learn from labeled data (e.g., spam detection)
  • Unsupervised Learning: Find patterns in unlabeled data (e.g., customer segmentation)
  • Reinforcement Learning: Learn through trial and error (e.g., game AI)
# Simple Linear Regression Example
import numpy as np

# Data: House sizes (sq ft) and prices (lakhs)
X = np.array([600, 800, 1000, 1200, 1500])
y = np.array([30, 40, 50, 60, 75])

# Calculate slope (m) and intercept (b)
# Formula: y = mx + b
mean_x = np.mean(X)
mean_y = np.mean(y)

m = np.sum((X - mean_x) * (y - mean_y)) / np.sum((X - mean_x) ** 2)
b = mean_y - m * mean_x

print(f"Equation: y = {m:.4f}x + {b:.4f}")

# Predict price for 1100 sq ft house
new_size = 1100
predicted_price = m * new_size + b
print(f"Predicted price for {new_size} sq ft: ‚Çπ{predicted_price:.2f} lakhs")

# Output:
# Equation: y = 0.0500x + 0.0000
# Predicted price for 1100 sq ft: ‚Çπ55.00 lakhs

Problem 1: Two Sum

Easy

Problem: Given an array of integers and a target, find two numbers that add up to the target.

Example:

Input: arr = [2, 7, 11, 15], target = 9

Output: [0, 1] (because arr[0] + arr[1] = 2 + 7 = 9)

Constraints:

  • Array length: 2 to 10,000
  • Each element: -10^9 to 10^9
  • Exactly one solution exists

üí° Solution Approach

Method 1: Brute Force (O(n¬≤))

function twoSum(arr, target) {
    for (let i = 0; i < arr.length; i++) {
        for (let j = i + 1; j < arr.length; j++) {
            if (arr[i] + arr[j] === target) {
                return [i, j];
            }
        }
    }
}

Method 2: Hash Map (O(n)) - Optimal!

function twoSum(arr, target) {
    let map = new Map();
    
    for (let i = 0; i < arr.length; i++) {
        let complement = target - arr[i];
        
        if (map.has(complement)) {
            return [map.get(complement), i];
        }
        
        map.set(arr[i], i);
    }
}

// Example
console.log(twoSum([2, 7, 11, 15], 9)); // [0, 1]

Output: "olleh"

// Method 1: Built-in methods
function reverseString(str) {
    return str.split('').reverse().join('');
}

// Method 2: Two pointers
function reverseString(str) {
    let arr = str.split('');
    let left = 0;
    let right = arr.length - 1;
    
    while (left < right) {
        [arr[left], arr[right]] = [arr[right], arr[left]];
        left++;
        right--;
    }
    
    return arr.join('');
}

console.log(reverseString("hello")); // "olleh"

Problem 3: Maximum Subarray Sum

Medium

Problem: Find the contiguous subarray with the largest sum.

Example:

Input: [-2, 1, -3, 4, -1, 2, 1, -5, 4]

Output: 6 (subarray [4, -1, 2, 1])

üí° Kadane's Algorithm

function maxSubArray(arr) {
    let maxSum = arr[0];
    let currentSum = arr[0];
    
    for (let i = 1; i < arr.length; i++) {
        // Either extend existing subarray or start new one
        currentSum = Math.max(arr[i], currentSum + arr[i]);
        maxSum = Math.max(maxSum, currentSum);
    }
    
    return maxSum;
}

// Example
let arr = [-2, 1, -3, 4, -1, 2, 1, -5, 4];
console.log(maxSubArray(arr)); // 6

// Step by step:
// i=0: current=-2, max=-2
// i=1: current=1, max=1 (start new)
// i=2: current=-2, max=1
// i=3: current=4, max=4 (start new)
// i=4: current=3, max=4
// i=5: current=5, max=5
// i=6: current=6, max=6 ‚úì
// i=7: current=1, max=6
// i=8: current=5, max=6
